CREATE TABLE USERS
(
    ID   SERIAL PRIMARY KEY,
    FIRSTNAME   VARCHAR(100),
    LASTNAME VARCHAR(100)

CREATE TABLE POSTS
(
    ID  SERIAL PRIMARY KEY,
    USER_ID  BIGINT UNSIGNED NOT NULL,
    BODY   VARCHAR(1024),
    FOREIGN KEY (USER_ID) REFERENCES USERS(ID)
);

FOREIGN KEY (ISSUESLIST_ID) REFERENCES ISSUESLISTS(ID),
